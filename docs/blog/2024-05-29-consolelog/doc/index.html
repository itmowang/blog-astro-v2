console.log

常规的打印

console.log("魔王");

console.warm

警告的打印

console.warn("魔王");

console.error

错误的打印

console.error("魔王");

console.dir

输出对象

 const obj = { name: "魔王", age: 999 };
 console.dir(obj);

console.clear()

清空控制台

console.clear();

console.table

以表格的形式输出对象或者数组

const test = ["Apple", "Banana", "Orange"];
    const test1 = {
      name: "mowang",
      age: 999,
    };
    console.table(test);
    console.table(test1);

console.assert()

用于条件不满足的时候输出信息

console.assert(2 + 2 === 5, "2+2不等于5");
